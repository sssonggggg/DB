
---merge 사용법
DECLARE @SEQ INT
SELECT @SEQ = ISNULL(MAX(SEQ),0)+1 FROM SEQ1
--SELECT @SEQ = 23
MERGE INTO SEQ1 AS A
USING( SELECT @SEQ AS SEQ) AS B
ON ( A.SEQ = B.SEQ )
WHEN MATCHED THEN
UPDATE SET A.SEQ = @SEQ, A.COL2 = '6' , A.COL3 = '12343'
WHEN NOT MATCHED  THEN
INSERT (SEQ, COL2, COL3) VALUES ((SELECT ISNULL(MAX(SEQ),0)+1 FROM SEQ1) , '6' , '123123');
